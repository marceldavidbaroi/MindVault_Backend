@Entity("people")
@Index(["userId"])
export class Person {
  @PrimaryGeneratedColumn("uuid")
  id: string;

  @Column({ type: "uuid" })
  userId: string;

  // Identity
  @Column()
  firstName: string;

  @Column({ nullable: true })
  lastName?: string;

  @Column({ nullable: true })
  nickname?: string;

  // Contact
  @Column({ nullable: true })
  primaryEmail?: string;

  @Column({ nullable: true })
  primaryPhone?: string;

  @Column({ nullable: true })
  alternateEmail?: string;

  @Column({ nullable: true })
  alternatePhone?: string;

  // Location
  @Column({ nullable: true })
  city?: string;

  @Column({ nullable: true })
  country?: string;

  @Column({ nullable: true })
  addressLine1?: string;

  @Column({ nullable: true })
  addressLine2?: string;

  // Classification
  @Column({
    type: "enum",
    enum: ["friend", "family", "colleague", "client", "acquaintance", "other"],
    default: "other",
  })
  type: string;

  @Column({
    type: "enum",
    enum: ["low", "medium", "high", "critical"],
    default: "medium",
  })
  importanceLevel: string;

  // Special Dates (for M4 Notifications)
  @Column({ type: "date", nullable: true })
  birthDate?: string;

  @Column({ type: "date", nullable: true })
  anniversaryDate?: string;

  @Column({ type: "date", nullable: true })
  workAnniversaryDate?: string;

  @Column({ type: "date", nullable: true })
  relationshipStartDate?: string;

  // Social Handles (hard coded)
  @Column({ nullable: true })
  instagram?: string;

  @Column({ nullable: true })
  facebook?: string;

  @Column({ nullable: true })
  linkedin?: string;

  @Column({ nullable: true })
  discord?: string;

  @Column({ nullable: true })
  skype?: string;

  // JSONB Metadata
  @Column({ type: "jsonb", default: {} })
  metadata: Record<string, any>;

  // Soft Delete
  @Column({ type: "timestamp", nullable: true })
  deletedAt?: Date;

  @CreateDateColumn()
  createdAt: Date;

  @UpdateDateColumn()
  updatedAt: Date;
}
